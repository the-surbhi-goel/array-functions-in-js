Existing shift

    const arr1 = [1, 2, 3, 4];
    arr1.shift();


Custom push

    Array.prototype.myShift = function () {
        let len = this.length;
        for (let i = 1; i < len; i++) {
            this[i - 1] = this[i];
        }
        this.length = this.length - 1;
    };
    
    const arr2 = [1, 2, 3, 4];
    arr2.myShift();
